43class JoinGroup(Sequence): 

44 """A group of scaffolds to be joined together 

45 

46 This class represents a group of scaffolds that should be joined 

47 together as specified in a joins file. Optionally, a group can be given 

48 a name to use for the new superscaffold instead of using 

49 `make_superscaffold_name()` to make one up based on the names of the 

50 components. If accessed as a `Sequence`, it provides direct access to 

51 the underlying list of scaffolds, e.g., 

52 

53 >>> from agp.join import JoinGroup 

54 >>> join_group = JoinGroup(["scaffold_1", "-scaffold_2", "+scaffold_3"]) 

55 >>> len(join_group) 

56 3 

57 >>> join_group[1] 

58 '-scaffold_2' 

59 

60 Attributes: 

61 scaffolds: a list of scaffolds to join together, in the correct 

62 order. A scaffold name can be preceded by "+" or "-" to 

63 indicate orientation; if none is specified, "+" is assumed. 

64 name: a name to use for the new superscaffold in the output agp 

65 """ 

66 

67 scaffolds: List[str] 

68 name: Optional[str] = None 

69 

70 def __getitem__(self, index) -> str: 

71 return self.scaffolds[index] 

72 

73 def __len__(self) -> int:

77def joins_type(filename: str) -> List[JoinGroup]: 

78 """ 

79 argparse type function for file listing scaffold joins 

80 

81 Args: 

82 filename: filename giving path to a file listing 

83 scaffolds to join. Each line should contain a comma- 

84 separated list of scaffolds to be joined into a single 

85 scaffold. Optionally, the file can contain a second column 

86 after a tab that gives each group a name. 

87 

88 Returns: 

89 list of join groups. Scaffold names may have "+" or "-" 

90 before their name; that is the job of downstream code to handle. 

91 

92 Raises: 

93 ScaffoldUsedTwiceError: if a scaffold is used multiple times 

94 """ 

95 joins: List[JoinGroup] = [] 

96 with open(filename) as joins_file: 

97 for line in joins_file: 

98 if not empty_line_regex.match(line): 

99 columns = line.strip().split("\t") 

100 join_group = JoinGroup(columns[0].split(",")) 

101 if len(columns) > 1: 

102 if not sequence_name_regex.match(columns[1]): 

103 raise BadSequenceNameError(columns[1]) 

104 join_group.name = columns[1] 

105 joins.append(join_group) 

106 

107 # look for scaffolds used more than once 

108 scaffold_counts = Counter(s.lstrip("+-") for s in chain.from_iterable(joins)) 

109 reused_scaffolds = [scaf for scaf, count in scaffold_counts.items() if count > 1] 

110 if reused_scaffolds: 

111 raise ScaffoldUsedTwiceError(f"Scaffolds used 2+ times: {reused_scaffolds}") 

112 

113 return joins

116def make_superscaffold_name(subscaffold_names: Iterable[str]) -> str: 

117 """ 

118 Comes up with a nice superscaffold name based on the given 

119 subscaffold names. If the subscaffold names are all in the format 

120 '[prefix]_[suffix]', where prefix is the same for all subscaffolds, 

121 then the superscaffold name is '[prefix]_[suffix1]p[suffix2]p[etc.]' 

122 Otherwise, it's the names of all scaffolds concatenated with 'p'. 

123 

124 Args: 

125 subscaffold_names (list(str)): list of names of subscaffolds 

126 being combined into a superscaffold 

127 

128 Returns: 

129 superscaffold_name (str): a name for the new scaffolds based 

130 on the subscaffold names 

131 """ 

132 # TODO this casting is necessary but ugly as sin. Find a nicer way 

133 matches = list(map(scaffold_regex.search, subscaffold_names)) 

134 if all(matches): 

135 if all( 

136 cast(Match, m).group(1) == cast(Match, matches[0]).group(1) for m in matches 

137 ): 

138 prefix = cast(Match, matches[0]).group(1) 

139 suffix = "p".join([cast(Match, m).group(2) for m in matches]) 

140 return "{}_{}".format(prefix, suffix) 

141 return "p".join(subscaffold_names)

144def join_scaffolds( 

145 superscaffold_rows: List[List[AgpRow]], 

146 gap_size: int = 500, 

147 gap_type: str = "scaffold", 

148 gap_evidence: str = "na", 

149 name: str = None, 

150) -> List[AgpRow]: 

151 """ 

152 Transforms agp rows from multiple scaffolds into agp rows for a 

153 single superscaffold containing all the given scaffolds in the 

154 specified order. Scaffolds should be oriented properly before 

155 using this function. 

156 

157 Args: 

158 superscaffold_rows: a list of lists of agp rows. Each sub-list is a 

159 list of all rows in a single scaffold. These scaffolds will be 

160 joined in the order given 

161 gap_size: length of the new gaps created by joining scaffolds 

162 together 

163 gap_type: what to call the new gaps in the agp file 

164 gap_evidence: evidence type for linkage across gap 

165 name: name of new superscaffold. If None, this function will come 

166 up with a name based on the names of subscaffolds contained by 

167 this superscaffold 

168 

169 Returns: 

170 a list of AgpRow instances containing the new scaffold specification 

171 """ 

172 # make a nice name for the new superscaffold we are creating 

173 subscaffold_names = (s[0].object for s in superscaffold_rows) 

174 if name is None: 

175 superscaffold_name = make_superscaffold_name(subscaffold_names) 

176 else: 

177 superscaffold_name = name 

178 

179 # keeps track of what component number of the superscaffold we are 

180 # currently on 

181 component_number_counter = 1 

182 # keeps track of the ending position of the previous subscaffold, 

183 # so that we can add it as an offset to components of the current 

184 # subscaffold 

185 end_of_previous_scaffold = 0 

186 # list(AgpRow) containing the rows to be returned 

187 outrows = [] 

188 # loop over the subscaffolds 

189 for i, this_scaffold_rows in enumerate(superscaffold_rows): 

190 # loop over the agp rows in this subscaffold 

191 for row in this_scaffold_rows: 

192 # update the current row and put it in the output list 

193 row.object = superscaffold_name 

194 row.part_number = component_number_counter 

195 component_number_counter += 1 

196 row.object_beg += end_of_previous_scaffold 

197 row.object_end += end_of_previous_scaffold 

198 outrows.append(row) 

199 end_of_previous_scaffold = outrows[-1].object_end 

200 

201 # add a gap if we're not on the last subscaffold 

202 if i < len(superscaffold_rows) - 1: 

203 outrows.append( 

204 GapRow( 

205 superscaffold_name, 

206 end_of_previous_scaffold + 1, 

207 end_of_previous_scaffold + gap_size, 

208 component_number_counter, 

209 length=gap_size, 

210 gap_type=gap_type, 

211 evidence=gap_evidence, 

212 ) 

213 ) 

214 component_number_counter += 1 

215 end_of_previous_scaffold = outrows[-1].object_end 

216 

217 return outrows

220def run( 

221 joins_list: List[JoinGroup], 

222 outfile: TextIO, 

223 agp_infile: Iterator[Union[str, AgpRow]], 

224 gap_size: int = 500, 

225 gap_type: str = "scaffold", 

226 gap_evidence: str = "na", 

227): 

228 """ 

229 Runs the 'join' module of agptools. 

230 """ 

231 # first, make a dict mapping the names of all scaffolds that will 

232 # be modified to an empty list which will later contain all agp rows 

233 # of that scaffold. 

234 scaffolds_to_join: Dict[str, List[AgpRow]] = {} 

235 for name in (s.lstrip("+-") for s in chain.from_iterable(joins_list)): 

236 scaffolds_to_join[name] = [] 

237 

238 # print all the rows to be output as-is and put the rows that need 

239 # to be modified first into the correct slot of the 

240 # scaffolds_to_join dict 

241 for row in agp_infile: 

242 if isinstance(row, AgpRow) and row.object in scaffolds_to_join: 

243 scaffolds_to_join[row.object].append(row) 

244 else: 

245 print(row, file=outfile) 

246 

247 # make sure we found agp entries for all the scaffolds 

248 for scaffold_name, rows in scaffolds_to_join.items(): 

249 if len(rows) == 0: 

250 raise ScaffoldNotFoundError(f"Scaffold {scaffold_name} not found in agp.") 

251 

252 # loop through each join group 

253 for i, join_group in enumerate(joins_list): 

254 scaffold_rows = [] 

255 # loop through the scaffolds in this join group 

256 for scaffold_name in join_group: 

257 if scaffold_name.startswith("-"): 

258 # reverse-complement if necessary 

259 scaffold_rows.append( 

260 reverse_rows(scaffolds_to_join[scaffold_name.lstrip("-")]) 

261 ) 

262 else: 

263 scaffold_rows.append(scaffolds_to_join[scaffold_name.lstrip("+")]) 

264 

265 # print out all the rows for this join group 

266 for row in join_scaffolds( 

267 scaffold_rows, gap_size, gap_type, name=join_group.name 

268 ): 

269 print(row, file=outfile)
