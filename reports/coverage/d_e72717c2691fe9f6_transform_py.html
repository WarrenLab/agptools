43def create_contig_dict(agp_in: Iterator[Union[str, AgpRow]]) -> ContigDict: 

44 """ 

45 Load the agp file into a dictionary mapping contig name to a list 

46 of rows in which that contig is the component_id. 

47 

48 Args: 

49 agp_in: an agp file to read into a dictionary 

50 

51 Returns: 

52 a dict mapping contig name to a list of rows in which that 

53 contig is the component_id 

54 """ 

55 contig_dict: ContigDict = defaultdict(list) 

56 for row in (r for r in agp_in if isinstance(r, AgpRow) and not r.is_gap): 

57 contig_dict[row.component_id].append(row) 

58 return contig_dict

61def transform_single_position(position: int, agp_row: AgpRow) -> int: 

62 """Transform a position from component to object coordinates 

63 

64 Given a position in component coordinates (e.g., a position on a 

65 contig), and the agp row containing that position, transform the 

66 position to object coordinates (e.g., the same position, but on 

67 a scaffold of which that contig is a part). 

68 

69 Args: 

70 position: a position in component/contig coordinates 

71 agp_row: the row containing that position 

72 

73 Returns: 

74 the same position, but in object/scaffold coordinates 

75 """ 

76 

77 # first, offset the position by where it is on this _row_. This 

78 # is necessary because not all rows contain an entire component/ 

79 # contig. E.g., if this row contains positions 101-200 on the 

80 # contig, then position 150 on the contig is the 50th base in 

81 # the piece of the contig enclosed by this row 

82 position = position - agp_row.component_beg + 1 

83 if agp_row.orientation == "+": 

84 # then, offset the position by where the row is on the object/ 

85 # scaffold 

86 position += agp_row.object_beg - 1 

87 elif agp_row.orientation == "-": 

88 position = agp_row.object_end - position + 1 

89 else: 

90 raise BadOrientationError(f"Orientation must be +/-: {agp_row}") 

91 return position

94def find_agp_row( 

95 contig_name: str, coordinate_on_contig: int, contig_dict: ContigDict 

96) -> AgpRow: 

97 """Find the agp row containing a coordinate. 

98 

99 Given a dictionary mapping contig name to a list of agp rows 

100 containing that contig, and a single position in contig coordinates, 

101 find and return the AGP row containing that position. 

102 

103 Args: 

104 contig_name: 

105 the contig containing the coordinate 

106 coordinate_on_contig: 

107 the position to look for, in contig coordinates 

108 contig_dict: 

109 a dictionary mapping contig name to a list of agp rows 

110 containing that contig 

111 

112 Returns: 

113 a single AGP row containing the requested position 

114 """ 

115 contig_list = contig_dict[contig_name] 

116 if not contig_list: 

117 raise NoSuchContigError(contig_name) 

118 for agp_row in contig_list: 

119 if ( 

120 agp_row.component_beg <= coordinate_on_contig 

121 and agp_row.component_end >= coordinate_on_contig 

122 ): 

123 return agp_row 

124 

125 raise CoordinateNotFoundError(f"{contig_name}:{coordinate_on_contig}")

128def transform_bed_row(bed_row: BedRange, contig_dict: ContigDict) -> BedRange: 

129 """Transform a bed row to scaffold coordinates 

130 

131 Transform a single row of a bed file from contig coordinates to 

132 scaffold coordinates. 

133 

134 Args: 

135 bed_row: the row to transform 

136 contig_dict: 

137 a dictionary mapping contig name to the agp row(s) 

138 containing that contig 

139 

140 Returns: 

141 the same bed row, but in scaffold coordinates now 

142 """ 

143 if bed_row.start is None or bed_row.end is None: 

144 raise UnsupportedOperationError( 

145 f"Transforming coordinateless bed row: {bed_row}" 

146 ) 

147 

148 # retrieve the AGP rows containing the bed range's start and end points 

149 agp_row_start = find_agp_row(bed_row.chrom, bed_row.start, contig_dict) 

150 agp_row_end = find_agp_row(bed_row.chrom, bed_row.end, contig_dict) 

151 

152 # only transform bed ranges that are entirely within one scaffold 

153 if agp_row_start.object == agp_row_end.object: 

154 bed_row.chrom = agp_row_start.object 

155 # transform the start and end positions of the bed range to the 

156 # new coordinate system 

157 bed_row.start, bed_row.end = sorted( 

158 [ 

159 transform_single_position(bed_row.start, agp_row_start), 

160 transform_single_position(bed_row.end, agp_row_end), 

161 ] 

162 ) 

163 # flip the orientation of the bed row, if necessary 

164 if ( 

165 agp_row_start == agp_row_end 

166 and bed_row.strand 

167 and agp_row_start.orientation == "-" 

168 ): 

169 if bed_row.strand == "+": 

170 bed_row.strand = "-" 

171 else: 

172 bed_row.strand = "+" 

173 else: 

174 raise UnsupportedOperationError( 

175 f"Transforming bed row spanning multiple scaffolds: {bed_row}" 

176 ) 

177 

178 return bed_row

182 bed_in: Iterator[BedRange], agp_in: Iterator[Union[str, AgpRow]], bed_out: TextIO 

183): 

184 contig_dict = create_contig_dict(agp_in)
